## 🚀 4. Production-Grade Implementation

```csharp
public class RankingService : IRankingService
{
    private readonly RankingWeights _weights;

    public RankingService(RankingWeights weights)
    {
        _weights = weights;
    }

    public List<RankingResult> RankCandidates(JobProfile job, List<CandidateProfile> candidates, int topN = 10)
    {
        var results = new List<RankingResult>(candidates.Count);

        Parallel.ForEach(candidates, candidate =>
        {
            var score = ComputeScore(job, candidate);

            lock (results)
            {
                results.Add(new RankingResult
                {
                    CandidateId = candidate.Id,
                    Score = score
                });
            }
        });

        return results
            .OrderByDescending(r => r.Score)
            .Take(topN)
            .ToList();
    }

    private double ComputeScore(JobProfile job, CandidateProfile candidate)
    {
        double semantic = CosineSimilarity(job.Embedding, candidate.Embedding);
        double skill = SkillMatch(job.RequiredSkills, candidate.Skills);
        double experience = ExperienceMatch(job.MinYearsOfExperience, candidate.YearsOfExperience);
        double keyword = KeywordBoost(job.RawText, candidate.RawText);

        return (_weights.SemanticWeight * semantic)
             + (_weights.SkillWeight * skill)
             + (_weights.ExperienceWeight * experience)
             + (_weights.KeywordWeight * keyword);
    }

    private double CosineSimilarity(float[] a, float[] b)
    {
        if (a == null || b == null || a.Length != b.Length) return 0;

        double dot = 0, magA = 0, magB = 0;

        for (int i = 0; i < a.Length; i++)
        {
            dot += a[i] * b[i];
            magA += a[i] * a[i];
            magB += b[i] * b[i];
        }

        return dot / (Math.Sqrt(magA) * Math.Sqrt(magB) + 1e-10);
    }

    private double SkillMatch(List<string> required, List<string> candidate)
    {
        if (required.Count == 0) return 0;

        var matchCount = required.Intersect(candidate, StringComparer.OrdinalIgnoreCase).Count();
        return (double)matchCount / required.Count;
    }

    private double ExperienceMatch(int required, int actual)
    {
        if (actual >= required) return 1.0;

        return (double)actual / required;
    }

    private double KeywordBoost(string jobText, string resumeText)
    {
        if (string.IsNullOrEmpty(jobText) || string.IsNullOrEmpty(resumeText))
            return 0;

        var keywords = jobText.Split(' ', StringSplitOptions.RemoveEmptyEntries)
                              .Distinct()
                              .Take(20);

        int hits = keywords.Count(k => resumeText.Contains(k, StringComparison.OrdinalIgnoreCase));

        return (double)hits / keywords.Count();
    }
}
```
